<script>
    document.addEventListener("DOMContentLoaded", () => {
    let currentAudio = null; // Variable to track the currently playing audio

    document.querySelectorAll('.play-buttons-container a').forEach(link => {
        link.addEventListener('click', function(e) {
            e.preventDefault();
            var audio = document.getElementById(link.querySelector('img').getAttribute('data-audio'));
            var img = link.querySelector('img');

            // If there is a currently playing audio and it's not the same as the one being clicked, pause it
            if (currentAudio && currentAudio !== audio) {
                currentAudio.pause();
                currentAudio.currentTime = 0;
                let currentImg = document.querySelector(`img[data-audio="${currentAudio.id}"]`);
                if (currentImg) {
                    currentImg.src = "/img/2024-frontpage/play-button.png"; // Change to the path of your play button image
                }
                currentAudio = null;
            }

            // Add event listener for when the audio ends
            audio.addEventListener('ended', function() {
                img.src = "/img/2024-frontpage/play-button.png";
                currentAudio = null;
            });

            // Play or pause the audio and update the icon
            if (audio.paused) {
                audio.play();
                img.src = "/img/2024-frontpage/pause-button.png"; // Change to the path of your pause button image
                currentAudio = audio; // Set the current audio to the one being played
            } else {
                audio.pause();
                img.src = "/img/2024-frontpage/play-button.png"; // Change to the path of your play button image
                currentAudio = null; // Reset current audio if paused
            }
        });
    });
});

<script>
    document.addEventListener("DOMContentLoaded", () => {
    let currentAudio = null; // Variable to track the currently playing audio

    document.querySelectorAll('.play-buttons-container a').forEach(link => {
        link.addEventListener('click', function(e) {
            e.preventDefault();
            var audio = document.getElementById(link.querySelector('img').getAttribute('data-audio'));
            var img = link.querySelector('img');

            // If there is a currently playing audio and it's not the same as the one being clicked, pause it
            if (currentAudio && currentAudio !== audio) {
                currentAudio.pause();
                currentAudio.currentTime = 0;
                let currentImg = document.querySelector(`img[data-audio="${currentAudio.id}"]`);
                if (currentImg) {
                    currentImg.src = "/img/2024-frontpage/play-button.png"; // Change to the path of your play button image
                }
                currentAudio = null;
            }

            // Add event listener for when the audio ends
            audio.addEventListener('ended', function() {
                img.src = "/img/2024-frontpage/play-button.png";
                currentAudio = null;
            });

            // Play or pause the audio and update the icon
            if (audio.paused) {
                audio.play();
                img.src = "/img/2024-frontpage/pause-button.png"; // Change to the path of your pause button image
                currentAudio = audio; // Set the current audio to the one being played
            } else {
                audio.pause();
                img.src = "/img/2024-frontpage/play-button.png"; // Change to the path of your play button image
                currentAudio = null; // Reset current audio if paused
            }
        });
    });
});

<script>
    document.addEventListener("DOMContentLoaded", () => {
    let currentAudio = null; // Variable to track the currently playing audio

    document.querySelectorAll('.play-buttons-container a').forEach(link => {
        link.addEventListener('click', function(e) {
            e.preventDefault();
            var audio = document.getElementById(link.querySelector('img').getAttribute('data-audio'));
            var img = link.querySelector('img');

            // If there is a currently playing audio and it's not the same as the one being clicked, pause it
            if (currentAudio && currentAudio !== audio) {
                currentAudio.pause();
                currentAudio.currentTime = 0;
                let currentImg = document.querySelector(`img[data-audio="${currentAudio.id}"]`);
                if (currentImg) {
                    currentImg.src = "/img/2024-frontpage/play-button.png"; // Change to the path of your play button image
                }
                currentAudio = null;
            }

            // Add event listener for when the audio ends
            audio.addEventListener('ended', function() {
                img.src = "/img/2024-frontpage/play-button.png";
                currentAudio = null;
            });

            // Play or pause the audio and update the icon
            if (audio.paused) {
                audio.play();
                img.src = "/img/2024-frontpage/pause-button.png"; // Change to the path of your pause button image
                currentAudio = audio; // Set the current audio to the one being played
            } else {
                audio.pause();
                img.src = "/img/2024-frontpage/play-button.png"; // Change to the path of your play button image
                currentAudio = null; // Reset current audio if paused
            }
        });
    });
});
</script>
