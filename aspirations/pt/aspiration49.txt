var wpcom_remote_login_extra_auth = '';
function wpcom_remote_login_remove_dom_node_id( element_id ) {
	var dom_node = document.getElementById( element_id );
	if ( dom_node ) { dom_node.parentNode.removeChild( dom_node ); }
}
function wpcom_remote_login_remove_dom_node_classes( class_name ) {
	var dom_nodes = document.querySelectorAll( '.' + class_name );
	for ( var i = 0; i < dom_nodes.length; i++ ) {
		dom_nodes[ i ].parentNode.removeChild( dom_nodes[ i ] );
	}
}
function wpcom_remote_login_final_cleanup() {
	wpcom_remote_login_remove_dom_node_classes( "wpcom_remote_login_msg" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_validate" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_js" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_iframe" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_styles" );
}

// Watch for messages back from the remote login
window.addEventListener( "message", function( e ) {
	if ( e.origin === "https://r-login.wordpress.com" ) {
		var data = {};
		try {
			data = JSON.parse( e.data );
		} catch( e ) {
			wpcom_remote_login_final_cleanup();
			return;
		}

		if ( data.msg === 'LOGIN' ) {
			// Clean up the login check iframe
			wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );

			var id_regex = new RegExp( /^[0-9]+$/ );
			var token_regex = new RegExp( /^.*|.*|.*$/ );
			if (
				token_regex.test( data.token )
				&& id_regex.test( data.wpcomid )
			) {
				// We have everything we need to ask for a login
				var script = document.createElement( "script" );
				script.setAttribute( "id", "wpcom_remote_login_validate" );
				script.src = '/remote-login.php?wpcom_remote_login=validate'
					+ '&wpcomid=' + data.wpcomid
					+ '&token=' + encodeURIComponent( data.token )
					+ '&host=' + window.location.protocol
					+ '//' + window.location.hostname
					+ '&postid=5937'
					+ '&is_singular=1';
				document.body.appendChild( script );
			}

			return;
		}

		// Safari ITP, not logged in, so redirect
		if ( data.msg === 'LOGIN-REDIRECT' ) {
			window.location = 'https://wordpress.com/log-in?redirect_to=' + window.location.href;
			return;
		}

		// Safari ITP, storage access failed, remove the request
		if ( data.msg === 'LOGIN-REMOVE' ) {
			var css_zap = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } @media screen and ( max-width: 782px ) { html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } }';
			var style_zap = document.createElement( 'style' );
			style_zap.type = 'text/css';
			style_zap.appendChild( document.createTextNode( css_zap ) );
			document.body.appendChild( style_zap );

			var e = document.getElementById( 'wpcom_request_access_iframe' );
			e.parentNode.removeChild( e );

			document.cookie = 'wordpress_com_login_access=denied; path=/; max-age=31536000';

			return;
		}

		// Safari ITP
		if ( data.msg === 'REQUEST_ACCESS' ) {
			console.log( 'request access: safari' );

			// Check ITP iframe enable/disable knob
			if ( wpcom_remote_login_extra_auth !== 'safari_itp_iframe' ) {
				return;
			}

			// If we are in a "private window" there is no ITP.
			var private_window = false;
			try {
				var opendb = window.openDatabase( null, null, null, null );
			} catch( e ) {
				private_window = true;
			}

			if ( private_window ) {
				console.log( 'private window' );
				return;
			}

			var iframe = document.createElement( 'iframe' );
			iframe.id = 'wpcom_request_access_iframe';
			iframe.setAttribute( 'scrolling', 'no' );
			iframe.setAttribute( 'sandbox', 'allow-storage-access-by-user-activation allow-scripts allow-same-origin allow-top-navigation-by-user-activation' );
			iframe.src = 'https://r-login.wordpress.com/remote-login.php?wpcom_remote_login=request_access&origin=' + encodeURIComponent( data.origin ) + '&wpcomid=' + encodeURIComponent( data.wpcomid );

			var css = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 46px !important; } * html body { margin-top: 46px !important; } @media screen and ( max-width: 660px ) { html { margin-top: 71px !important; } * html body { margin-top: 71px !important; } #wpcom_request_access_iframe { display: block; height: 71px !important; } } #wpcom_request_access_iframe { border: 0px; height: 46px; position: fixed; top: 0; left: 0; width: 100%; min-width: 100%; z-index: 99999; background: #23282d; } ';

			var style = document.createElement( 'style' );
			style.type = 'text/css';
			style.id = 'wpcom_request_access_styles';
			style.appendChild( document.createTextNode( css ) );
			document.body.appendChild( style );

			document.body.appendChild( iframe );
		}

		if ( data.msg === 'DONE' ) {
			wpcom_remote_login_final_cleanup();
		}
	}
}, false );

		+ "&time=" + Math.floor( Date.now() / 1000 );
	document.body.appendChild( iframe );
}, false );

		<div class="entry-content" data-first_letter="L">
		<p style="text-align: left"><strong>[<em>La fantasía no es un fantasma</em>]</strong></p>
<p style="text-align: right">por <strong><a href="#_ftn1" name="_ftnref1">Anthony Sampson</a></strong></p>
<p style="text-align: right"><span id="more-5937"></span></p>
<p style="text-align: right">Tradução<span style="color: #ff6600"> |</span> <strong><a href="#_ftn2" name="_ftnref2">Paulo Sérgio de Souza Jr.</a></strong></p>
<blockquote><p><em>Sampson, A. (1992) “La fantasía no es un fantasma”, Artefacto, n. 3, México.</em></p></blockquote>
<blockquote>
<p style="text-align: center"><em>Longe de fazer com que as línguas se comuniquem, a tradução enfatiza o que as torna estranhas umas às outras.</em></p>
<p>*</p>
<p><em>Michel Cresta</em><a href="#_ftn3" name="_ftnref3"><sup>[1]</sup></a></p></blockquote>
<p><a href="#_ftn3" name="_ftnref3"></a></p>
<p style="text-align: justify">Numa atividade que vem se tornando cada vez mais crítica — e criticável —, resulta conveniente, até mesmo indispensável, explicitar certas opções feitas pelo tradutor. De fato, há quem tenha visto na tarefa de tradução nada além de uma série ininterrupta de escolhas<a href="#_ftn4" name="_ftnref4"><sup>[2]</sup></a>. Segundo essa tese, a essência da tradução consistiria no próprio ato de escolher. Talvez seja nessa primordialidade da escolha que o caráter ético da tarefa do tradutor, de acordo com Benjamin, possa ser visto mais facilmente<a href="#_ftn5" name="_ftnref5"><sup>[3]</sup></a>.</p>
<p style="text-align: justify">Seja como for, temos obrigação de explicar nossa escolha por “fantasia” para traduzir o francês <em>fantasme</em>, e não “fantasma”, como costumam fazer alguns.</p>
<p style="text-align: justify">Antes de mais nada, vamos estabelecer alguns pontos incontroversos. <em>Fantasme</em>, em francês, é o termo especificamente psicanalítico empregado — pelo menos originalmente, desde a década de 1920 — para traduzir o termo alemão <em>Phantasie</em>. Esse termo, exatamente como o inglês <em>fantasy</em> ou o espanhol <em>fantasía</em>, embora possua uma gama de sentidos ampla, designa basicamente a imaginação: “Não tanto a faculdade de imaginar”, dizem Laplanche e Pontalis em seu <em>Vocabulário da psicanálise</em>, “no sentido filosófico do termo (<em>Einbildungskraft</em>), mas o mundo imaginário, os seus conteúdos, a atividade criadora que o anima (<em>das Phantasieren</em>). Freud retomou essas diferentes acepções da língua alemã”<a href="#_ftn6" name="_ftnref6"><sup>[4]</sup></a>. Laplanche e Pontalis continuam explicando que, em francês, o termo <em>fantasme</em> foi revivido pelos analistas, fato que explica por que suas ressonâncias psicanalíticas são maiores que as do vocábulo <em>Phantasie</em> e por que ele possui uma extensão menor que o homólogo alemão. Eles também lembram que Lagache propôs a reutilização do termo <em>fantaisie</em> em seu sentido antigo, “mas que, para a consciência linguística, contemporânea, [esse termo] dificilmente pode deixar de sugerir os matizes de capricho, originalidade, ausência de seriedade etc.”</p>
<p style="text-align: justify">Noutros termos, <em>fantasme</em> é o vocábulo ao qual os primeiros tradutores das obras de Freud tiveram obrigatoriamente de recorrer. Porque a palavra <em>fantaisie</em> — que teria sido o equivalente natural —, devido à evolução específica da língua francesa, era simplesmente inutilizável, uma vez que carregava ideias de leviandade, de liberdade imaginativa, até mesmo de sorte ou felicidade, absolutamente incompatíveis com o caráter constritivo ou coercitivo da <em>Phantasie</em> freudiana (aliás, observamos que, de acordo com o Diccionario crítico etimológico de la lengua castellana [Dicionário crítico etimológico da língua castelhana], de J. Corominas, foi esse sentido “italiano” de “fantasia” como capricho, vontade caprichosa, que Juan de Valdés — autor do Diálogo de la lengua española [Diálogo da língua espanhola], e falecido em 1541 — quis aclimatar em espanhol, mas sem sucesso).</p>
<p style="text-align: justify">Assim, os franceses não tinham à disposição imediata uma palavra com a mesma carga semântica que <em>fantasy</em>, em inglês, ou <em>fantasía</em>, em espanhol. Em francês, uma frase como <em>fantaisie de castration</em> (capricho de castração) ou <em>fantaisie de meurtre</em> (capricho de assassinato) seria impensável; surgiria um puro contrassenso. De modo que os tradutores se viram compelidos a ressuscitar um termo antigo que havia caído em desuso desde o século XIII. Parece que essa escolha era tão óbvia e tão inevitável que não houve sequer necessidade de uma reunião da Commission Linguistique pour L’Unification du Vocabulaire Psychanalytique Français [Comissão Linguística para a Unificação do Vocabulário Psicanalítico Francês], cuja existência e funcionamento são documentados por Elisabeth Roudinesco em sua história da psicanálise na França<a href="#_ftn7" name="_ftnref7"><sup>[5]</sup></a>. Os tradutores escolheram um termo, então, cujo significado “original” — diferentemente do que acontece com o <em>fantasma</em> espanhol — se havia perdido por completo; e que era capaz, por seu caráter neutro, de receber qualquer sentido: de felicidade ou de infelicidade, material ou abstrato etc.</p>
<p style="text-align: justify">Mas desde a sua introdução, como aponta Guy Le Gaufey no seminário <em>De la réalité du fantasme </em>[<em>A realidade da fantasia</em>]<a href="#_ftn8" name="_ftnref8"><sup>[6]</sup></a>, esse feliz achado dos tradutores teve um sucesso inaudito: deixou de ser um termo meramente técnico num vocabulário específico e restrito, e passou a ser uma palavra extremamente corrente na língua cotidiana. Hoje, portanto, em francês, <em>fantasme</em> é uma expressão que pode aparecer em qualquer conversa ou em qualquer leitura, sem que a pessoa que a emprega esteja necessariamente ciente de sua origem psicanalítica.<a href="#_ftn9" name="_ftnref9"><sup>[7]</sup></a></p>
<p style="text-align: justify">Vamos deixar claro o seguinte, então: em francês, <em>fantasme</em> não é um termo próprio de Lacan e dos lacanianos. Ele foi introduzido para traduzir <em>Phantasie</em>, e hoje em dia é pura e simplesmente o francês que todos os psicanalistas de língua francesa escrevem quando pretendem falar do que Freud chamou de <em>Phantasie</em>, quaisquer que sejam suas posições doutrinárias. Em termos ainda mais claros, acharemos <em>phantasme</em> nos textos de Sacha Nacht, Benassy, Bouvet, Levobici, Diatkine, Laplanche, assim como nos textos de Lacan e dos seus alunos. Não tem absolutamente nada de específico quanto à indicação de uma determinada posição teórica.</p>
<p style="text-align: justify">De modo geral, os tradutores dos textos dos primeiros autores mencionados (Nacht, Diatkine etc.) para o espanhol se deram conta disso, pois não costumam traduzir <em>fantasme</em> por <em>fantasma</em>. E os tradutores ingleses, tanto dos primeiros citados quanto de Lacan, também se deram conta disso: traduzem por <em>fantasy</em>, que é, nem mais nem menos, o único termo equivalente ao alemão <em>Phantasie</em>. E decerto não recorreram a <em>ghost</em> para traduzir <em>fantasme</em> quando na pena de Lacan!</p>
<p style="text-align: justify">Consideramos que <em>fantasme</em>, em francês, e <em>fantasma</em>, em espanhol, são o que os franceses chamam de <em>faux amis </em>(literalmente, “falsos amigos”): termos que, por sua semelhança fonética, induzem ao engano os iniciantes no estudo de uma língua estrangeira, pois seu aparente ar de familiaridade produz a impressão de uma equivalência de sentido que, na realidade, não existe.</p>
<p style="text-align: justify">Com efeito, apostaríamos que os primeiros leitores de Lacan hispanofalantes, deslumbrados com a extraordinária novidade do ensino lacaniano, tomaram como certo que <em>fantasme</em> só poderia ser mais uma das inovações e neologismos de Lacan. Pensamos que muitos nunca se deram conta dessa equivocação, e que outros, os quais talvez se tenham dado conta disso, tomaram o partido dessa tradução com a finalidade de transformar <em>fantasma</em> numa espécie de cavalo de batalha para tentar, por meio de um léxico, estabelecer distâncias em relação a grupos e doutrinas não lacanianos. Nesse ponto, aludimos especialmente aos kleinianos — cujos textos apelam em massa para a noção de “fantasia” —, tão abundantes, em particular no Cone Sul.</p>
<p style="text-align: justify">A luta dos lacanianos nesses países para implementar a renovação lacaniana da clínica desperta a nossa simpatia e a nossa solidariedade. No entanto, se o objetivo é conferir vida nova a um conceito, modificando seu emprego e seu conteúdo, é completamente inútil abandonar o termo existente e recorrer a um novo, cujo sentido atual — resultado de um longo processo histórico — se pretende eliminar à canetada.</p>
<p style="text-align: justify">A reformulação lacaniana do conceito freudiano de “fantasia” passa pela elucidação de sua lógica, e não pela cunhagem de um novo vocábulo, o que, como vimos, Lacan nunca realizou, contentando-se com o termo francês <em>fantasme</em>, comum a todos os psicanalistas de língua francesa.</p>
<p style="text-align: justify">Suspeitamos que, para muitos, “fantasma” nada mais é que uma espécie de criptografia, um modo de reconhecer-se entre si como antagonistas de outros. Assim, seu uso se reduz, na melhor das hipóteses, a um santo-e-senha<a href="#_ftn10" name="_ftnref10"><sup>[8]</sup></a>; e, na pior, a um crachá de congresso.</p>
<p style="text-align: justify"><em>Fantasma</em> é um termo de uso muito antigo em espanhol; é atestado desde o século XIII:</p>
<p style="text-align: justify;padding-left: 40px"><em>En el Apocalisi Sant Juán Evangelis­ta / Non vido tal figura nin espantable vista; / En grand hato daríe gran lucha é grand conquista; / Non sé de quál diablo es tal fantasma quista. </em>(Arcipreste de Hita, Libro de buen amor, 1011).</p>
<p style="text-align: justify">Só séculos mais tarde <em>fantasma</em> mudou de gênero, de feminino para masculino, pois mesmo em 1770 Ramón de la Cruz intitula o seu sainete <em>La fantasma del lugar </em>[<em>A fantasma do lugar</em>]<a href="#_ftn11" name="_ftnref11"><sup>[9]</sup></a>. Mas igualmente antigo é o termo <em>fantasía</em>:</p>
<p style="text-align: justify;padding-left: 40px"><em>Fiziéronse las gentes todas maravilladas, / tenién que fantasía las avié engannadas, / pero a pocca d’ora fueron certifigadas / rendién gracias a Christo todas manos alzadas. </em>(Gonzalo de Berceo, Milagros de Nuestra Señora [El parto maravilloso], 443).</p>
<p style="text-align: justify">Portanto, os dois termos coexistiram ininterruptamente desde os primórdios do idioma. Além disso, nenhum deles jamais caiu em desuso ou foi confundido com o outro. Eles tiveram uma existência paralela e independente, com a manutenção de seus sentidos específicos sem nunca se sobrepor ou baralhar.</p>
<p style="text-align: justify">O Diccionario de la Real Academia [<em>Dicionário da Real Academia</em>] registra quatro acepções para o vocábulo <em>fantasma</em>:</p>
<p style="text-align: justify;padding-left: 40px">(a) Visão quimérica, como a proporcionada por sonhos ou pela acalorada imaginação; (b) Imagem de um objeto que fica impressa na fantasia; (c) Pessoa entojada, sisuda e presunçosa; (d) Abantesma ou pessoa fantasiada que sai à noite para assustar as gentes.</p>
<p style="text-align: justify">Não há dúvida de que o uso atual privilegia as acepções (a) e (d). Basta ler os jornais e a literatura contemporânea para verificar. Manchete do <em>La Prensa</em> (Bogotá) de 28/3/1990: “<em>Fantasmas rondan en Casaloma</em>” [Fantasmas assombram Casaloma]. Ou no mesmo jornal, em 25/9/1990, a crônica de folhetim proveniente de Valverde de Leganés, na Espanha, em que “os vizinhos vivem em estado de alerta há dias por causa das andanças de um ‘fantasma’ que, segundo a Guarda Civil, sai ‘quase todas as noites’”; e, num nível muito diferente, o livro de poemas de Luis Vidales, <em>El Libro de los fantasmas</em> [<em>O livro dos fantasmas</em>] (os exemplos poderiam ser multiplicados indefinidamente — sem necessidade alguma de trazer à baila aquele personagem mascarado de uma conhecidíssima história em quadrinhos).</p>
<p style="text-align: justify">Que exemplo de “fantasma” seria melhor que o espectro do pai de Hamlet — acepção (a) — ou o pai de Ofelia? — acepção (c): “pessoa entojada, sisuda e presunçosa”. Vamos dizer, então, que o <em>pai</em> é, por definição, um fantasma? Talvez seja, mas não descartemos tão rapidamente a distinção freudiana entre a <em>imago</em> do pai e a <em>função paterna</em>, em sua decisiva incidência sobre a fantasia, conforme a estuda Freud, por exemplo, em <em>Uma criança apanha</em><a href="#_ftn12" name="_ftnref12"><sup>[10]</sup></a>.</p>
<p style="text-align: justify">A <em>Enciclopedia del idioma: diccionario histórico y moderno de la lengua española (Siglos XII al XX) etimológico, tecnológico, regional e hispanoamericano </em>[<em>Enciclopédia do idioma: dicionário histórico e moderno da língua espanhola (séculos XII a XX), etimológico, tecnológico, regional e hispano-americano</em>], de Martín Alonso, acrescenta mais alguns significados aos quatro do <em>Diccionario de la Academia </em>[<em>Dicionário da Academia</em>]:</p>
<p style="text-align: justify;padding-left: 40px">(e) Na terminologia escolástica, imagem sensível, individual e concreta das coisas, em oposição a uma ideia geral e abstrata: f. TELEV. Aparecimento gradual da imagem na tela, passando da escuridão total à plena visibilidade [visivelmente, tradução de <em>ghost</em> (fantasma), e equivalente televisivo do <em>fading </em>radiofônico], g. Pl. S. XIII, ocasion., Berceo: S. Dom., 70; h. F. magnético. FÍS. Figura obtida quando se jogam limalhas de ferro sobre uma folha de papel colocada no campo magnético de um ímã, que se agrupam formando figuras que determinam as linhas de força do campo magnético mencionado.</p>
<p style="text-align: justify">A tudo isso, a Enciclopedia universal ilustrada europeo-ame­ricana [<em>Enciclopédia universal ilustrada euroamericana</em>], da editora Espasa-Calpe, volume XXIII, acrescenta uma outra acepção:</p>
<p style="text-align: justify;padding-left: 40px">estátua de madeira, na qual os cirurgiões praticam a aplicação de bandagens em operações de parto.</p>
<p style="text-align: justify">No entanto, a parte central do artigo que a referida enciclopédia dedica ao conceito de <em>fantasma</em> gira em torno dos espectros e da crença universal no retorno das almas dos mortos após o falecimento. Ele termina remetendo às publicações contemporâneas da Society for Psychical Research [Sociedade de Pesquisas Psíquicas] — da qual Jung foi membro — sobre os Phantasms of the living [Os fantasmas dos vivos] e sua relação com a telepatia<a href="#_ftn13" name="_ftnref13"><sup>[11]</sup></a>.</p>
<p style="text-align: justify">O testemunho do <em>Diccionario crítico etimológico de la lengua castellana</em>, de J. Corominas, também é inequívoco: no verbete <em>Fantasía</em>, faz menção a <em>fantasma</em> (certificando seu emprego desde Berceo, Juan Ruiz e Nebrija) definido como “aparição”, “imagem”, “espectro”.</p>
<p style="text-align: justify">De modo que a coisa pareceria bastante clara: desde o Arcipreste de Hita até os dias de hoje, <em>fantasma</em>, exceto por raros usos ocasionais e técnicos, quer dizer “espectro”. Observemos também, para o leitor não familiarizado com o idioma francês, que o <em>fantasma</em> espanhol, em francês, não se diz <em>fantasme</em>, mas <em>fantôme</em> — <em>ghost</em>, em inglês, e <em>Phantom</em>, em alemão. E em italiano, de igual maneira, <em>fantasma</em> tem o sentido de espectro:</p>
<p style="text-align: justify;padding-left: 40px"><em>Volgarmente si chiamano fantasmi le anime dei morti che ritornano o s’aggirano sulla terra sotto la forma di scheletri o di corpi umani avvolti in un aereo velo</em>.<a href="#_ftn14" name="_ftnref14"><sup>[12]</sup></a></p>
<p style="text-align: justify">Parece-nos, por isso mesmo, completamente inútil buscar “um sentido não comum” (<em>Dicionário de filosofia</em> de José Ferrater Mora) de “fantasma” como <em>produto</em>: “as imagens produzidas pela fantasia”. Essa distinção fantasia/fantasma (<em>phantasía</em>/phántasma) é uma distinção gnoseológica que remonta a Aristóteles, a quem deve suas cartas nobiliárquicas, e dele provém essa problemática em torno da qual, nas palavras de Lacan, a filosofia ocidental tem girado desde então<a href="#_ftn15" name="_ftnref15"><sup>[13]</sup></a>.</p>
<p style="text-align: justify">Porque essa problemática, da qual o par fantasia/fantasma é inseparável, não é outra senão a própria problemática da representação. Ferrater Mora traduz <em>phantásmata</em> (plural, em grego, de phántasma) precisamente como “representações”, e em seu <em>Dicionário</em> nos ensina que a sistematização desse par de termos complementares se deve a Aristóteles, para quem a fantasia está relacionada à apetência, aos movimentos apetitivos em direção a algo desejado, ao passo que os <em>phantásmata</em> (fantasmas) seriam as representações em potência ou “ideias” atualizáveis por meio das percepções. Desse modo, “a fantasia reside em nosso poder de suscitar (‘conjurar’) imagens mesmo quando os objetos ou fontes de sensações não se achem imediatamente presentes”<a href="#_ftn16" name="_ftnref16"><sup>[14]</sup></a>. Mas qual é, então, o estatuto gnoseológico da fantasia e de seu produto, o fantasma? Eles podem ser objeto de crença e convicção, ou são apenas criações de um espírito ocioso que se diverte com suas imaginações improdutivas? Desde o início, portanto, com essa problemática inaugurada por Aristóteles, nós nos encontramos inevitavelmente imersos em toda a lodosa questão do verdadeiro e do falso, da ilusão e da alucinação, da percepção fiel e da percepção falaciosa. Foram inúmeras as reelaborações e soluções propostas, desde Crisipo, o estoico — que introduziu finas distinções entre a <em>phantasía</em>, o <em>phantastón</em>, o <em>phantastikón</em> e o phántasma<a href="#_ftn17" name="_ftnref17"><sup>[15]</sup></a> —, até Sartre e os fenomenólogos<a href="#_ftn18" name="_ftnref18"><sup>[16]</sup></a>. Porém, por mais apaixonantes que sejam as análises dos metafísicos, antigos e modernos, Freud não se inscreve nessa tradição filosófica ocidental<a href="#_ftn19" name="_ftnref19"><sup>[17]</sup></a> — ao menos se a opinião de Lacan merece de nós algum crédito.</p>
<p style="text-align: justify">Na sessão de 16 de dezembro de 1959 do seu seminário <em>A ética da psicanálise</em>, Lacan examina em detalhe a natureza da “representação”, a <em>Vorstellung</em> alemã:</p>
<p style="text-align: justify;padding-left: 40px">O caráter de composição imaginária, de elemento imaginário do objeto, faz dele o que se poderia chamar de a substância da aparência, o material de um engodo vital, uma aparição sujeita à decepção de uma <em>Erscheinung</em> [aparição, espectro, fantasma, segundo o <em>Langenscheidts Handwörterbuch Deutsch-Spanisch</em>], diria eu se me permitisse falar alemão, isto é, aquilo em que a aparência se sustenta, mas que é igualmente a aparição da assombração, a aparição corriqueira [&#8230;]. A <em>Vorstellung</em> é algo de essencialmente descomposto. É aquilo em torno do qual gira, desde sempre, a filosofia do Ocidente, desde Aristóteles, e a <em>phantasía.</em></p>
<p style="text-align: justify;padding-left: 40px">Em Freud a <em>Vorstellung</em> é apreendida em seu caráter radical sob a forma pela qual é introduzida numa filosofia que é essencialmente traçada pela teoria do conhecimento. E é isso que é notável — ele lhe designa, até ao extremo, o caráter ao qual os filósofos, precisamente, não puderam decidir-se a reduzi-la, o de um corpo vazio, de um fantasma [enfatizo, <em>fantôme</em>, no original francês], de um pálido íncubo da relação com o mundo, de um gozo extenuado que constitui seu traço essencial através de toda a interrogação do filósofo. E isolando-a nessa função, Freud a extrai da tradição.<a href="#_ftn20" name="_ftnref20"><sup>[18]</sup></a></p>
<p style="text-align: justify">Assim, se Freud extrai o conceito de “representação” (<em>Vorstellung</em>) de toda a tradição filosófica ocidental, atribuindo-lhe o caráter de um “fantasma”, “pálido íncubo da relação com o mundo”, não seria aconselhável admitir que, precisamente, a sua “fantasia” (<em>Phantasie</em>) também carece de todo e qualquer vínculo com a metafísica — onde se emparelhava com seu correlato, o “fantasma” —, seja ela qual for (aristotélica, estoica, epicurista, tomista&#8230; ou qualquer outra)?</p>
<p style="text-align: justify">Desde o início, desde o <em>Projeto</em> de 1895, o próton pseudos<a href="#_ftn21" name="_ftnref21"><sup>[19]</sup></a> deixará de fazer parte da teoria do silogismo e, como primeira mentira, será a maneira pela qual o sujeito, no nível do inconsciente, diz a primeira verdade; a “fantasia” freudiana não terá mais com a verdade a mesma relação que a problemática metafísica do par indissolúvel “fantasia/fantasma”. A <em>Phantasie</em> freudiana é outra coisa, e é por isso que não vamos encontrar seu sentido especificado em nenhum <em>Wörterbuch</em><a href="#_ftn22" name="_ftnref22"><sup>[20]</sup></a>.</p>
<p style="text-align: justify">Contudo, <em>Phantasie</em> tem uma gama tão ampla de sentidos quanto <em>fantasía</em>, em espanhol — que, de acordo com o <em>Diccionario de la Real Academia</em> [<em>Dicionário da Real Academia</em>], tem ainda mais significados que <em>fantasma</em>. Mas cumpre saber que, se Freud quisesse ter escrito <em>Phantasma</em> (“fantasma”), poderia ter feito, já que esse vocábulo também existe em alemão — embora de uso reconhecidamente escasso fora do contexto da metafísica e da medicina psiquiátrica<a href="#_ftn23" name="_ftnref23"><sup>[21]</sup></a> —, mas, como quase sempre acontece com Freud, ele preferiu um termo comum do idioma e descartou o vocábulo pouco frequente. Mas, em contrapartida, esse termo <em>Phantasma</em> foi empregado por alguém próximo de Freud, ninguém menos que&#8230;. Josef Breuer, em <em>Estudos sobre a histeria</em>, no célebre histórico clínico da “Srta. Anna O”!<a href="#_ftn24" name="_ftnref24"><sup>[22]</sup></a> E praticamente não há necessidade de lembrar que Freud havia assistido — como deveria ser de conhecimento comum — aos cursos de Frantz Brentano, o grande comentador de Aristóteles e teórico da representação e da fantasia<a href="#_ftn25" name="_ftnref25"><sup>[23]</sup></a>, e nos quais Freud necessariamente se familiarizaria com o uso do par aristotélico fantasia/fantasma. Diante dessas duas referências contemporâneas, que sentido tem invocar — como fazem alguns para justificar a tradução de <em>fantasme</em> por “fantasma” — a função propriamente retórica da “dêixis em fantasma” (<em>Deixis am Phantasma</em>), proposta por K. Bühler, que se produz “quando um narrador leva o ouvinte para o reino do ausente recordável, ou para o reino da fantasia construtiva, e ali o presenteia com os mesmos demonstrativos para que ele possa ver e ouvir o que há ali para ver e ouvir (e tocar, como se sabe, e talvez também cheirar e saborear)?<sup> <a href="#_ftn26" name="_ftnref26"><sup>[24]</sup></a></sup></p>
<p style="text-align: justify">Além disso, pode-se sustentar que empregar na psicanálise o conceito <em>metafísico</em> de “fantasma” — por oposição ao seu sentido <em>comum</em> de “espectro” — seria, sem dúvida, privilegiar excessivamente o produto-imagem-objeto às custas da atividade, da “ficção” que a produz. A fantasia freudiana não se reduz a ser a imagem pura e extenuada do objeto. Isso seria desconhecer a sua estrutura gramatical — sua estrutura de relato, estrutura narrativa que implica um sujeito, um verbo e um predicado. Daí apresentar-se como um palco, um roteiro, uma encenação, com atores e libretos. Por certo, o objeto não poderia estar ausente em semelhante encenação (a escrita lacaniana, sem dúvida, salientou isso), mas tampouco se deve esquecer que justamente o objeto <em>a</em> não é suscetível a nenhuma representação especular.</p>
<p style="text-align: justify">Por todas essas razões, então, não é de surpreender que <em>todos</em> os tradutores de Freud para o espanhol — desde Luis López Ballesteros y de Torre, passando por Ludovico Rosenthal até José Luis Etcheverry — não tenham hesitado, por um instante sequer, em traduzir <em>Phantasie</em> por <em>fantasía</em>, exatamente como, por razões que já explicitamos, todos os tradutores franceses traduziram o mesmo termo alemão por <em>fantasme</em> (<em>Phantasie</em> = <em>fantasme</em> = <em>fantasía</em> = “fantasia”, ou na ordem que se bem quiser).</p>
<p style="text-align: justify">Parece-nos totalmente inverossímil<a href="#_ftn27" name="_ftnref27"><sup>[25]</sup></a> e incongruente com a posição freudiana toda e qualquer tentativa de traduzir por “fantasma” quando se trata, supostamente, da fantasia originária ou inconsciente propriamente dita, e por “fantasia” quando se trata da fantasia pré-consciente ou do sonho diurno. Sabe-se que, tanto em inglês quanto em francês, há quem tenha tentado introduzir ortograficamente essa distinção, escrevendo phantasy ou phantasme no primeiro caso, e fantasy ou fantasme no segundo. Essas tentativas não prosperaram e, até onde sabemos, Lacan nunca as aprovou ou imitou. Há uma incontornável ambiguidade no emprego freudiano do vocábulo <em>Phantasie</em>, uma oscilação constante que é o que permite a Freud todas as suas análises dos processos criativos nos artistas, as formações das memórias encobridoras, os ataques histéricos, as compulsões obsessivas e também suas especulações acerca das origens da cultura humana etc.; por isso, nenhuma grafia, nenhum truque lexical pode absolver o leitor de sua distração ou desinteresse.</p>
<p style="text-align: justify">Por último, então, reservamos para o final o argumento que nos parece mais decisivo: se o termo em espanhol, quando se trata de Freud, é <em>fantasía</em>, não poderia haver um termo distinto quando se trata de Lacan. Isso é muito claro e contundente. Se há dois, um para Freud e outro para Lacan, um dos dois fica sobrando. Na nossa opinião, sobra <em>fantasma</em>.<span style="color: #ff6600">♦</span></p>

<p style="text-align: justify"><a href="#_ftnref24" name="_ftn24"><sup>[22]</sup></a> “Durante toda a evolução da doença, os dois estados de consciência coexistiram: o primário, em que a paciente era de todo normal psiquicamente, e o “segundo” estado, que bem podemos comparar ao sonho, em função de sua riqueza em fantasmas [<em>Phantasme</em>] e alucinações”. BREUER, Josef, “Srta. Anna O.”. In: FREUD, Sigmund (1893-95) <em>Obras completas</em>, vol. 2: “Estudos sobre a histeria”. Trad. L. Barreto. São Paulo: Companhia das Letras, 2016, pp. 72-73; trad. modificada.</p>
<p style="text-align: justify"><a href="#_ftnref25" name="_ftn25"><sup>[23]</sup></a> Cf. BRENTANO, Frantz (1874) “Psicología (Psicología desde el punto de vista empírico)” [Trad. J. Gaos], Revista de Occidente, Madrid, 1926.</p>
<p style="text-align: justify"><a href="#_ftnref26" name="_ftn26"><sup>[24]</sup></a> CARRETER, Fernando Lázaro. Diccionario de términos filológicos. Madrid: Credos, 1968, p. 130.</p>
<p style="text-align: justify"><a href="#_ftnref27" name="_ftn27"><sup>[25]</sup></a> Nessa inverosimilhança incorre o tradutor do <em>Vocabulário</em> de Laplanche e Pontalis para o espanhol. O original francês indica que <em>fantasme</em> se traduz por <em>fantasía</em>, em espanhol, mas o Dr. Fernando Cervantes Gimeno, professor associado de Psicologia Médica na Faculdade de Medicina de Barcelona, prefere realizar a complicada manobra, que o próprio Freud nunca quis fazer, de oscilar entre “fantasma” e “fantasia”, na vã esperança de, por fim, ancorar o navio-fantasma nas profundezas do inconsciente, enquanto a fantasia pode ser deixada flutuando à deriva da imaginação totalmente livre. Isso o leva a traduzir, por exemplo, assim: “Parece, então, que a problemática freudiana do fantasma (<em>sic</em>) não apenas não permite efetuar uma distinção de natureza entre fantasma (<em>sic</em>) inconsciente e fantasia (<em>sic</em>) consciente, mas tende a apontar suas analogias, suas relações estreitas, os passos entre elas”. Somos obrigados a nos perguntar se o tradutor lê o que ele traduz.</p>
